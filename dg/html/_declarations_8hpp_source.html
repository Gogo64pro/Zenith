147
148 std::stringstream ss;
149 for (const auto& ann : annotations) {
150 ss << pad << ann->toString() << "\n";
151 }
152 ss << pad << accessNames[flags.access]
153 << (flags.isConst ? " CONST " : " ")
154 << (flags.isStatic ? " STATIC " : " ")
155 << kindNames[flags.kind] << " " << name;
156 if (type) ss << " : " << type->toString();
157 for (size_t i = 0; i < initializers.size(); ++i) {
158 if (i > 0) ss << ", ";
159 ss << initializers[i].first;
160 if (!initializers[i].first.empty()) ss << " = ";
161 ss << initializers[i].second->toString();
162 }
163 if (body) {
164 ss << " " << removePadUntilNewLine(body->toString(indent));
165 } else if (flags.kind == FIELD) {
166 ss << ";";
167 }
168 return ss.str();
