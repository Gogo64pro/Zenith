45 // --- Binary Operations ---
46 struct BinaryOpNode : ExprNode {
47 enum Op : uint8_t { ADD, SUB, MUL, DIV, EQ, NEQ, LT, GT, LTE, GTE, ASN, MOD, ADD_ASN, SUB_ASN, MUL_ASN, DIV_ASN, MOD_ASN } op;
48 std::unique_ptr<ExprNode> left;
49 std::unique_ptr<ExprNode> right;
50
51 BinaryOpNode(SourceLocation loc, TokenType tokenType,
52 std::unique_ptr<ExprNode> leftExpr,
53 std::unique_ptr<ExprNode> rightExpr)
54 : ExprNode(), op(convertTokenType(tokenType)),
55 left(std::move(leftExpr)),
56 right(std::move(rightExpr)) {
57 this->loc = loc;
58 }
59
60 std::string toString(int indent = 0) const override {
61 static const char* opNames[] = {"+", "-", "*", "/", "==", "!=", "<", ">", "<=", ">=", "=", "%", "+=", "-=", "*=", "/=", "%=", };
62 std::string pad(indent, ' ');
63 return pad + "BinaryOp(" + opNames[op] + ")\n" +
64 left->toString(indent + 2) + "\n" +
65 right->toString(indent + 2);
66 }
67
68 private:
69 static Op convertTokenType(TokenType type) {
70 switch(type) {
71 case TokenType::PLUS: return ADD;
72 case TokenType::MINUS: return SUB;
73 case TokenType::STAR: return MUL;
74 case TokenType::SLASH: return DIV;
75 case TokenType::EQUAL_EQUAL: return EQ;
76 case TokenType::BANG_EQUAL: return NEQ;
77 case TokenType::LESS: return LT;
78 case TokenType::GREATER: return GT;
79 case TokenType::LESS_EQUAL: return LTE;
80 case TokenType::GREATER_EQUAL: return GTE;
81 case TokenType::EQUAL: return ASN;
82 case TokenType::PERCENT: return MOD;
83 case TokenType::PLUS_EQUALS: return ADD_ASN;
84 case TokenType::MINUS_EQUALS: return SUB_ASN;
85 case TokenType::STAR_EQUALS: return MUL_ASN;
86 case TokenType::SLASH_EQUALS: return DIV_ASN;
87 case TokenType::PERCENT_EQUALS: return MOD_ASN;
88 default: throw std::invalid_argument("Invalid binary operator");
89 }
90 }
91 };

148 std::vector<std::pair<std::string, std::unique_ptr<ExprNode>>>&& props)
149 : ExprNode() {
150 this->loc = loc;

301 StructInitializerNode(SourceLocation loc, std::vector<StructFieldInitializer> fields)
302 : fields(std::move(fields)) {
303 this->loc = loc;
304 // Determine if this is purely positional
305 isPositional = true;
306 for (const auto& field : this->fields) {
307 if (!field.name.empty()) {

314 std::string pad(indent, ' ');
315 std::stringstream ss;
316 if(isPositional) ss << "Positional";
317 ss << "{";
318 for(size_t i=0; i<fields.size(); ++i){
319 if(i!=0) ss << ", ";
320 if(!fields[i].name.empty()) ss << fields[i].name << " : ";
321 ss << fields[i].value->toString(indent + 2);
322 }
323 ss << "}";
