123 bool first = true;
124 for (const auto& param : parameters) {
125 if (!first) ss << ", ";
126 first = false;
127
128 if (param.isVariadic) ss << "...";
129
130 switch (param.kind) {
131 case TemplateParameter::TYPE:
132 ss << "typename " << param.name;
133 if (param.defaultType) {
134 ss << " = " << param.defaultType->toString();
135 }
136 break;
137 case TemplateParameter::NON_TYPE:
138 ss << param.type->toString() << " " << param.name;
139 if (param.defaultValue) {
140 ss << " = " << param.defaultValue->toString();
141 }
142 break;
143 case TemplateParameter::TEMPLATE:
144 ss << "template<...> typename " << param.name;
145 break;
146 }
147 }
148
149 ss << ">\n";
